/*\

Macro to convert a list to Japanese comma saparated string

<<列表转句子 text:"* List to be converted" >>

<$macrocall $name=列表转句子 text={{我的奇妙清单}} />

\*/

(function() {
  exports.name = '列表转句子';

  exports.params = [{ name: 'text', defalue: '' }];

  exports.run = function(text) {
    return text
      .split(/\s?\*\s/)
      .filter(it => it)
      .join('、');
  };
})();
